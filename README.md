Usage:
# Install dependencies
npm install

# Install client dependencies
cd client
npm install
cd ..
# Install React
cd client
npm i react-admin ra-data-simple-rest @material-ui/core
cd ..
# Run client and server (Visit http://localhost:3000)
npm run dev

# Run server only (:5000)
npm run server

# Run client only (:3000)
npm run client
